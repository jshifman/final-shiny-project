#3-pointer stats:
# TP - 3 point fields made, TPA - 3 point attempts, TPP - 3 point percentage

#1980 TP & TPA from the NBA championship
TPA_and_TP <- function(dataset) {
  ret <- list()
  ret$TPA <- dataset %>%
    select(Year, TPA) %>%
    filter(Year == 1980)  %>%
    select(TPA) %>%
    summarise(total = sum(TPA)) %>%
    pull()
  ret$TP <-dataset %>%
    select(Year, TP) %>%
    filter(Year == 1980)  %>%
    select(TP) %>%
    summarise(total = sum(TP)) %>%
    pull()
  return (ret)
}
